function update(shouldSave = true) { let totalNow = 0; const assets = []; document.querySelectorAll('#assets-list .asset-item').forEach(item => { const val = parseFloat(item.querySelector('input').value) || 0; totalNow += val; assets.push({ name: item.getAttribute('data-name'), val: val }); // Индивидуальный ИИ под строкой const hint = item.querySelector('.ai-small'); if (val > 100000) hint.innerText = "🤖 Мощный актив, так держать!"; else if (val > 0) hint.innerText = "🤖 Резерв в безопасности"; else hint.innerText = "🤖 Пора бы пополнить этот счет"; }); const inc = parseFloat(document.getElementById('inc').value) || 0; const exp = parseFloat(document.getElementById('exp').value) || 0; const net = inc - exp; const totalFuture = totalNow + net; document.getElementById('r1').innerText = totalNow.toLocaleString() + " ₽"; document.getElementById('r2').innerText = totalFuture.toLocaleString() + " ₽"; // Анализ плана document.getElementById('ai-inc').innerText = `🤖 Это ${Math.round(inc/30).toLocaleString()} ₽ в день`; document.getElementById('ai-exp').innerText = exp > inc * 0.8 ? "🚨 Опасно! Тратишь почти всё" : "🤖 Расходы под контролем"; // --- ГЛАВНЫЙ ВЫВОД ИИ (Логика из Obsidian) --- let mainAdvice = ""; const survivalMonths = exp > 0 ? (totalNow / exp).toFixed(1) : 0; if (net > 0) { mainAdvice = `📈 Капитал растет! За месяц ты прибавишь **${net.toLocaleString()} ₽**. `; if (survivalMonths >= 6) { mainAdvice += `Твоя подушка безопасности на высоте: ты проживешь **${survivalMonths} мес.** без работы. Время инвестировать!`; } else if (survivalMonths >= 3) { mainAdvice += `У тебя есть база на **${survivalMonths} мес.** Это неплохо, но цель — полгода.`; } else { mainAdvice += `Но запаса прочности почти нет (всего на **${survivalMonths} мес.**). Копи усерднее!`; } } else if (net < 0) { mainAdvice = `🚨 **Критическая ситуация!** Ты прожигаешь по **${Math.abs(net).toLocaleString()} ₽** в месяц. Твоих денег хватит всего на **${survivalMonths} мес.** Срочно режь косты!`; } else { mainAdvice = "⚖️ Ты в нуле. Деньги приходят и уходят. Чтобы богатеть, нужно либо поднять доход, либо снизить траты."; } document.getElementById('ai-main-advice').innerHTML = mainAdvice; // Приветствие и сохранение const name = tg.initDataUnsafe.user ? tg.initDataUnsafe.user.first_name : "Друг"; document.getElementById('user-greeting').innerText = `Привет, ${name}!`; if (shouldSave) { localStorage.setItem('myAssets', JSON.stringify(assets)); localStorage.setItem('myInc', inc); localStorage.setItem('myExp', exp); const indicator = document.getElementById('save-indicator'); indicator.style.opacity = "1"; setTimeout(() => { indicator.style.opacity = "0.4"; }, 1000); } }
